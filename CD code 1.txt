#include <stdio.h>
#include <string.h>

int main() {
    char sentence[1000];
    fgets(sentence, 1000, stdin); // read input sentence from user
    
    // tokenize the sentence by space
    char* token = strtok(sentence, " ");
    while (token != NULL) {
        printf("%s\n", token); // print each word in a new line
        token = strtok(NULL, " ");
    }
    
    return 0;
}
